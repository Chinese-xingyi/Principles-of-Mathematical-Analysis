\begin{Exercise}
\begin{proof}
Notice that $f(x) = g(x) = 0$.
We consider
$$
\lim_{t\to x}\frac{f(t)}{g(t)}
= \lim_{t\to x}\frac{f(t)-f(x)}{g(t)-g(x)}
= \lim_{t\to x}\left( \frac{f(t)-f(x)}{t-x} \cdot \frac{t-x}{g(t)-g(x)} \right)
= \frac{f'(x)}{g'(x)}.
$$
\end{proof}
\end{Exercise}
