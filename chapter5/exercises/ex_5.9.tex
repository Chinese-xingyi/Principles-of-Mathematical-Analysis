\begin{Exercise}
	\begin{proof}
		We have already known $\lim_{x\to 0}f'(x) = 3$.
		Suppose to contrary that $f'(0)\neq 3$, then $f'$ have a simple discontinuity at $x=3$, this contradicts the Corollary of Theorem 5.12.
		
		Hence $f'(0)$ exists and is equal to $3$.
	\end{proof}
\end{Exercise}
